.JOB GLSHRINK
.NOLOG
.NOTIME
* SHRINK THE GL D FILE
/  ENTER KEY NUMBER
.ENTER 9
.FSPEC GL#9D ON 1 4=SIZE
.SKIP TO NOGO IF 0 = #4
.NEW GL999 ON 1 'B' 5T
.USE GL999 ON 1 FOR SYS1
.USE GL#9D ON 1 FOR SYS0
.RUN S.XCOPUT ON 1
.SKIP 2 IF #C EQ 0
=ERR1
/********* COPY ERROR ********
.END
.DEL GL#9D 1
.NAM GL999 1 GL#9D
.END
/  FINISHED
S.CON
=NOGO

.END
